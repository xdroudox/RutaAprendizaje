Ejercicio 1: GET - Obtener posts

SOLUCION
function ejercicio1() { var resultado = document.getElementById('resultado-ej1-sol'); resultado.textContent = 'Cargando...'; fetch('https://jsonplaceholder.typicode.com/posts') .then(function(respuesta) { return respuesta.json(); }) .then(function(posts) { var primeros5 = posts.slice(0, 5); resultado.innerHTML = ''; primeros5.forEach(function(post) { var card = document.createElement('div'); card.className = 'card'; card.innerHTML = '<h3>' + post.title + '</h3><p>' + post.body + '</p>'; resultado.appendChild(card); }); }) .catch(function(error) { resultado.textContent = 'Error: ' + error.message; }); }
Explicacion: fetch(url) inicia la peticion GET. .then(r => r.json()) convierte la respuesta a objeto JS. slice(0, 5) toma los primeros 5 elementos. forEach itera y crea elementos HTML para cada post. .catch maneja errores de red.

Ejercicio 2: POST - Crear un recurso

SOLUCION
function ejercicio2() { var resultado = document.getElementById('resultado-ej2-sol'); var title = document.getElementById('input-title-sol').value.trim(); var body = document.getElementById('input-body-sol').value.trim(); if (!title || !body) { resultado.textContent = 'Completa el titulo y el cuerpo del post.'; return; } resultado.textContent = 'Creando post...'; fetch('https://jsonplaceholder.typicode.com/posts', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ title: title, body: body, userId: 1 }) }) .then(function(r) { return r.json(); }) .then(function(datos) { resultado.innerHTML = 'Post creado exitosamente!<br><br>' + JSON.stringify(datos, null, 2); }) .catch(function(error) { resultado.textContent = 'Error: ' + error.message; }); }
Explicacion: El segundo parametro de fetch incluye method: 'POST', headers con Content-Type, y body con los datos serializados a JSON. JSONPlaceholder simula la creacion y devuelve el objeto con un id (101).

Ejercicio 3: GET con manejo de errores

SOLUCION
async function ejercicio3() { var resultado = document.getElementById('resultado-ej3-sol'); var postId = document.getElementById('input-post-id-sol').value || '1'; resultado.textContent = 'Buscando post #' + postId + '...'; try { var respuesta = await fetch('https://jsonplaceholder.typicode.com/posts/' + postId); if (!respuesta.ok) { throw new Error('Post no encontrado (status ' + respuesta.status + ')'); } var post = await respuesta.json(); resultado.innerHTML = '<div class="card"><h3>' + post.title + '</h3>' + '<p>' + post.body + '</p>' + '<small>Post #' + post.id + ' | User #' + post.userId + '</small></div>'; } catch (error) { resultado.innerHTML = '<div style="color: #e74c3c;">Error: ' + error.message + '</div>'; } }
Explicacion: async/await simplifica el codigo asincrono. respuesta.ok verifica si el status esta entre 200-299. Si no es ok, se lanza un error con throw. El try/catch captura tanto errores HTTP como errores de red.
