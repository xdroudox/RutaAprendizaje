Ejercicio 1: Eventos de raton

SOLUCION
var caja1 = document.getElementById('caja-ej1-sol'); caja1.addEventListener('click', function() { caja1.style.background = '#3498db'; caja1.textContent = 'Click!'; logMsg('Click detectado'); }); caja1.addEventListener('mouseenter', function() { caja1.style.transform = 'scale(1.2)'; caja1.textContent = 'Hover!'; logMsg('Mouse enter'); }); caja1.addEventListener('mouseleave', function() { caja1.style.transform = 'scale(1)'; caja1.style.background = '#e74c3c'; caja1.textContent = 'Fuera!'; logMsg('Mouse leave'); });
Explicacion: addEventListener asocia funciones a eventos del elemento. click se dispara al hacer clic. mouseenter al entrar el raton. mouseleave al salir. La propiedad style modifica CSS inline.

Ejercicio 2: setTimeout y setInterval

SOLUCION
var intervaloCuenta = null; var contador = 5; function iniciarCuenta() { if (intervaloCuenta) return; contador = 5; document.getElementById('display-sol').textContent = contador; intervaloCuenta = setInterval(function() { contador--; if (contador <= 0) { document.getElementById('display-sol').textContent = 'Despegue!'; clearInterval(intervaloCuenta); intervaloCuenta = null; } else { document.getElementById('display-sol').textContent = contador; } }, 1000); } function detenerCuenta() { if (intervaloCuenta) { clearInterval(intervaloCuenta); intervaloCuenta = null; document.getElementById('display-sol').textContent = '-'; } }
Explicacion: setInterval(funcion, 1000) ejecuta la funcion cada 1000ms. Se decrementa el contador en cada ejecucion. Cuando llega a 0, se usa clearInterval para detenerlo. detenerCuenta() permite cancelar manualmente.

Ejercicio 3: Promises y async/await

SOLUCION
var modoErrorSol = false; function toggleErrorModeSol() { modoErrorSol = !modoErrorSol; var btn = document.getElementById('btn-error-sol'); btn.textContent = 'Modo error: ' + (modoErrorSol ? 'ON' : 'OFF'); btn.style.background = modoErrorSol ? '#e74c3c' : '#95a5a6'; } function esperar(ms) { return new Promise(function(resolve, reject) { setTimeout(function() { if (modoErrorSol) { reject(new Error('Error simulado')); } else { resolve(); } }, ms); }); } async function iniciarOperacionSol() { var displayOp = document.getElementById('display-op-sol'); displayOp.textContent = 'Cargando...'; displayOp.style.background = '#fff3cd'; try { await esperar(2000); displayOp.textContent = 'Operacion completada!'; displayOp.style.background = '#d4edda'; } catch (error) { displayOp.textContent = 'Error en la operacion'; displayOp.style.background = '#f8d7da'; } }
Explicacion: new Promise((resolve, reject) => {...}) crea una promesa. setTimeout dentro de la promesa simula una operacion asincrona. async/await permite escribir codigo asincrono como si fuera sincrono. try/catch maneja tanto errores de la promesa (reject) como excepciones.
